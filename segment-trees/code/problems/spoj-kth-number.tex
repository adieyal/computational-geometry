\begin{programmingproblem}{K-th Number}{1 second}{256 megabytes}{C}{prob:mkthnum}
    \problemdescription{
        You are working for Macrohard company in the data structures department. After failing your previous task about key insertion, you were asked to write a new data structure that would be able to quickly return the k-th order statistic in an array segment.\\
        That is, given an array $a[1 \ldots n]$ of different integer numbers, your program must answer a series of questions $Q(i, j, k)$ in the form: "What would be the k-th number in the $a[i \ldots j]$ segment, if this segment was sorted?"\\
        For example, consider the array $a = (1, 5, 2, 6, 3, 7, 4)$. Let the question be $Q(2, 5, 3)$. The segment $a[2 \ldots 5]$ is $(5, 2, 6, 3)$. If we sort this segment, we get $(2, 3, 5, 6)$, the third number is $5$, and therefore the answer to the question is $5$.
    }

    \probleminput{
        The first line contains two integers $n$ and $m$ ($1 \leq n \leq 100\,000$, $1 \leq m \leq 5\,000$) — the size of the array and the number of questions to answer.\\
        The second line contains $n$ different integer numbers not exceeding $10^9$ by their absolute values — the array for which the answers should be given.\\
        Each of the following $m$ lines contains a question description, each consisting of three integers: $i$, $j$, and $k$ ($1 \leq i \leq j \leq n$, $1 \leq k \leq j - i + 1$) and represents the question $Q(i, j, k)$.
    }

    \problemoutput{
        For each question, output the answer to it — the k-th number in the sorted $a[i \ldots j]$ segment.
    }

    \begin{problemexamples}
        \begin{example}{
            7 3\\
            1 5 2 6 3 7 4\\
            2 5 3\\
            4 4 1\\
            1 7 3
        }{
            5\\
            6\\
            3
        }
        \end{example}
    \end{problemexamples}

    \problemnotes{
        \textbf{Note:} A naive solution will not work for the given constraints!\\
        For the first query, the segment is $(5, 2, 6, 3)$, sorted: $(2, 3, 5, 6)$, the 3rd number is $5$.\\
        For the second query, the segment is $(6)$, the 1st number is $6$. \\
        For the third query, the segment is $(1, 5, 2, 6, 3, 7, 4)$, sorted: $(1, 2, 3, 4, 5, 6, 7)$, the 3rd number is $3$.
    }
\end{programmingproblem}
