\begin{programmingproblem}{DZY Loves Fibonacci Numbers}{4 seconds}{256 megabytes}{C}{prob:dzy}
    \problemdescription{
        In mathematical terms, the sequence $F_n$ of Fibonacci numbers is defined by the recurrence relation:
        \[
            F_1 = 1;\quad F_2 = 1;\quad F_n = F_{n-1} + F_{n-2}\quad (n > 2).
        \]
        DZY loves Fibonacci numbers very much. Today DZY gives you an array consisting of $n$ integers: $a_1, a_2, \ldots, a_n$. Moreover, there are $m$ queries, each query has one of the two types:
        \begin{itemize}
            \item Format of the query \texttt{1 l r}. In reply to the query, you need to add $F_{i-l+1}$ to each element $a_i$, where $l \leq i \leq r$.
            \item Format of the query \texttt{2 l r}. In reply to the query you should output the value of $\sum_{i=l}^{r} a_i$ modulo $10^9 + 9$.
        \end{itemize}
        Help DZY reply to all the queries.
    }
    
    \probleminput{
        The first line of the input contains two integers $n$ and $m$ ($1 \leq n, m \leq 300\,000$). The second line contains $n$ integers $a_1, a_2, \ldots, a_n$ ($1 \leq a_i \leq 10^9$) â€” initial array $a$.\\
        Then, $m$ lines follow. Each line describes a single query in the format given above. It is guaranteed that for each query, $1 \leq l \leq r \leq n$ holds.
    }
    
    \problemoutput{
        For each query of the second type, print the value of the sum on a single line.
    }
    
    \begin{problemexamples}
        \begin{example}{
            4\ 4\\
            1\ 2\ 3\ 4\\
            1\ 1\ 4\\
            2\ 1\ 4\\
            1\ 2\ 4\\
            2\ 1\ 3
        }{
            17\\
            12
        }
        \end{example}
    \end{problemexamples}
    
    \problemnotes{
        After the first query, $a = [2, 3, 5, 7]$.\\
        For the second query, sum $= 2 + 3 + 5 + 7 = 17$.\\
        After the third query, $a = [2, 4, 6, 9]$.\\
        For the fourth query, sum $= 2 + 4 + 6 = 12$.
    }
\end{programmingproblem}
