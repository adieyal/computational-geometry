\begin{programmingproblem}{DQUERY - D-query}{1 second}{256 megabytes}{C}{prob:dquery}
    \problemdescription{
        Given a sequence of $n$ numbers $a_1, a_2, \ldots, a_n$ and a number of d-queries.\\
        A d-query is a pair $(i, j)$ ($1 \leq i \leq j \leq n$). For each d-query $(i, j)$, you have to return the number of \term{distinct elements} in the subsequence $a_i, a_{i+1}, \ldots, a_j$.
    }
    
    \probleminput{
        The first line contains integer $n$ ($1 \leq n \leq 30000$).\\
        The second line contains $n$ numbers $a_1, a_2, \ldots, a_n$ ($1 \leq a_i \leq 10^6$).\\
        The third line contains integer $q$ ($1 \leq q \leq 200000$), the number of d-queries.\\
        Each of the next $q$ lines contains two integers $i$ and $j$ ($1 \leq i \leq j \leq n$), representing a d-query.
    }
    
    \problemoutput{
        For each d-query $(i, j)$, print the number of distinct elements in the subsequence $a_i, a_{i+1}, \ldots, a_j$ on a single line.
    }
    
    \begin{problemexamples}
        \begin{example}{
            5\\
            1\ 1\ 2\ 1\ 3\\
            3\\
            1\ 5\\
            2\ 4\\
            3\ 5
        }{
            3\\
            2\\
            3
        }
        \end{example}
    \end{problemexamples}
    
    \problemnotes{
        For the first query, the subsequence is $[1, 1, 2, 1, 3]$ which has $3$ distinct elements: $1, 2, 3$.\\
        For the second query, the subsequence is $[1, 2, 1]$ which has $2$ distinct elements: $1, 2$.\\
        For the third query, the subsequence is $[2, 1, 3]$ which has $3$ distinct elements: $1, 2, 3$.
    }
\end{programmingproblem}
