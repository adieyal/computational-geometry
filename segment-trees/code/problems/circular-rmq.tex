\begin{programmingproblem}{Circular RMQ}{1.5 seconds}{256 megabytes}{C}{prob:circularrmq}
    \problemdescription{
        You are given a circular array \sequence{a_0, a_1, \ldots, a_{n-1}}. There are two types of operations with it:
        \begin{itemize}
            \item \texttt{inc(lf, rg, v)} — this operation increases each element on the segment $[lf, rg]$ (inclusively) by $v$;
            \item \texttt{rmq(lf, rg)} — this operation returns the minimal value on the segment $[lf, rg]$ (inclusively).
        \end{itemize}
        Assume segments to be circular, so if $n = 5$ and $lf = 3$, $rg = 1$, it means the index sequence: $3, 4, 0, 1$.
        Write a program to process the given sequence of operations.
    }
    
    \probleminput{
        The first line contains integer $n$ ($1 \leq n \leq 200\,000$).\\
        The next line contains the initial state of the array: $a_0, a_1, \ldots, a_{n-1}$ ($-10^6 \leq a_i \leq 10^6$), $a_i$ are integers.\\
        The third line contains integer $m$ ($0 \leq m \leq 200\,000$), $m$ — the number of operations.\\
        Each of the next $m$ lines contains one operation.\\
        If a line contains two integers $lf, rg$ ($0 \leq lf, rg \leq n-1$), it means an \texttt{rmq} operation.\\
        If a line contains three integers $lf, rg, v$ ($0 \leq lf, rg \leq n-1$; $-10^6 \leq v \leq 10^6$), it means an \texttt{inc} operation.
    }
    
    \problemoutput{
        For each \texttt{rmq} operation, print the result for it on a single line.
    }
    
    \begin{problemexamples}
        \begin{example}{
            4\\
            1 2 3 4\\
            4\\
            3 0\\
            3 0 -1\\
            0 1\\
            2 1
        }{
            1\\
            0\\
            0
        }
        \end{example}
    \end{problemexamples}
    
    \problemnotes{
        For the first query, the segment is $[3, 0]$ which means elements $a_3, a_0$ (values $4, 1$), so the minimum is $1$.\\
        For the second query, increase $a_3$ and $a_0$ by $-1$, so the array becomes $[0, 2, 3, 3]$.\\
        For the third query, the segment is $[0, 1]$ (values $0, 2$), so the minimum is $0$.\\
        For the fourth query, the segment is $[2, 1]$ (values $3, 3, 0, 2$), so the minimum is $0$.
    }
\end{programmingproblem}
