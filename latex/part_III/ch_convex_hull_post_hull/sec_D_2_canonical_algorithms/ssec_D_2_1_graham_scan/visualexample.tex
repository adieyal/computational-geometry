        \begin{tikzpicture}[scale=1.2]

                \node[above right] at (\x,\y) {$p_{\n}$};
            }
            % Highlight p0
            \fill[red] (1,1) circle (3pt);
            % Draw arrows showing sorted order
            \draw[->, dashed, gray] (1,1) -- (3,1);
            \draw[->, dashed, gray] (3,1) -- (4,1);
